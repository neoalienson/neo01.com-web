class ResilientShardedDatabase {

    for (let attempt = 1; attempt <= maxRetries; attempt++) {
      try {
        return await this.queryShard(shard, query);
      } catch (error) {
        if (attempt === maxRetries) {
          // å¦‚æœå¯ç”¨ï¼Œå˜—è©¦å‰¯æœ¬

    }
  }
  
  async queryShard(shard, query) {

    return await connection.query(query);
  }
