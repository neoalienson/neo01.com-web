    // å¤åˆ¶æ¶ˆæ¯åˆ°æ‰€æœ‰è®¢é˜…è€…

      handler: handler

    const order = await this.db.orders.create(orderData);
    
    await this.broker.publish('orders', {
      type: 'OrderCreated',
      orderId: order.id,
      customerId: order.customerId,
      items: order.items,
      total: order.total,
      timestamp: new Date().toISOString()
    });
    
    return order;
  }
}
